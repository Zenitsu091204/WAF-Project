services:
  # --- SMTP Relay ---
  mail:
    image: ghcr.io/egos-tech/smtp:1.1.3
    restart: always

  # --- Caching and Queuing (Valkey/Redis) ---
  redis:
    image: valkey/valkey:7.2
    restart: always
    command: >
      sh -c "exec valkey-server --requirepass ${REDIS_PASSWORD}"
    volumes:
      - cache_data:/data
    healthcheck:
      test: ["CMD", "valkey-cli", "-a", "${REDIS_PASSWORD}", "ping"]
      interval: 5s
      timeout: 3s
      retries: 5

  # --- Main Database ---
  db:
    image: mariadb:10.11
    restart: always
    environment:
      MYSQL_USER: ${MYSQL_USER}
      MYSQL_PASSWORD: ${MYSQL_PASSWORD}
      MYSQL_ROOT_PASSWORD: ${MYSQL_ROOT_PASSWORD}
      MYSQL_DATABASE: ${MYSQL_DATABASE}
    volumes:
      - mysql_data:/var/lib/mysql
    healthcheck:
      test: ["CMD", "healthcheck.sh", "--connect", "--innodb_initialized"]
      interval: 10s
      timeout: 5s
      retries: 5

  # --- Expansion Modules ---
  misp-modules:
    image: ghcr.io/misp/misp-docker/misp-modules:latest
    restart: always

  # --- Core Application ---
  misp-core:
    image: ghcr.io/misp/misp-docker/misp-core:latest
    restart: always
    depends_on:
      redis:
        condition: service_healthy
      db:
        condition: service_healthy
    ports:
      - "8080:80"
      - "8443:443"
    volumes:
      - ./configs:/var/www/MISP/app/Config
      - ./logs:/var/www/MISP/app/tmp/logs
      - ./files:/var/www/MISP/app/files
      - ./ssl:/etc/nginx/certs
      - ./gnupg:/var/www/MISP/.gnupg
    environment:
      BASE_URL: "https://10.0.2.15:8443"
      ADMIN_EMAIL: ${ADMIN_EMAIL}
      ADMIN_PASSWORD: ${ADMIN_PASSWORD}
      ADMIN_ORG: ${ADMIN_ORG}
      MYSQL_HOST: db
      MYSQL_DATABASE: ${MYSQL_DATABASE}
      MYSQL_USER: ${MYSQL_USER}
      MYSQL_PASSWORD: ${MYSQL_PASSWORD}
      REDIS_HOST: redis
      REDIS_PORT: 6379
      REDIS_PASSWORD: ${REDIS_PASSWORD}
      ENABLE_DB_SETTINGS: "true"
      DISABLE_SSL_REDIRECT: "true" # Set to true since we handle 8443 externally
      DISABLE_IPV6: "true"
    healthcheck:
      test: ["CMD", "curl", "-sk", "https://localhost/users/heartbeat"]
      interval: 20s
      timeout: 10s
      retries: 10

volumes:
  mysql_data:
  cache_data:
